#include "stdio.h"
int main()&#123;
    int i,score,sum=0,avg;
    for(i=1;i<=3;i++)&#123;
        printf("请输入第%d名学生的成绩:\n",i);
        scanf("%d",&score);
        sum=sum+score;//累加求和
    &#125;
    avg=sum/3;//平均成绩
    printf("三名学的总成绩是:%d\n",sum);
    printf("平均成绩为:%d\n",avg);
